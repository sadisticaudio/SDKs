623 template<typename T>
624 void AAX_CParameter<T>::SetValue( T newValue )
625 {
626  double newNormalizedValue = mTaperDelegate->RealToNormalized(newValue);
627 
628  // <DMT> Always go through the automation delegate even if the control isn't automatable to prevent fighting with other GUIs.
629  // Somewhere back in the automation delegate, or elsewhere in the system, it will determine the differences in behavior surrounding
630  // automation. The only reason that there wouldn't be an automation delegate is if this parameter has yet to be added to a
631  // ParameterManager. Let's put the null value guards in place, just in case, and also for unit tests.
632  if ( mAutomationDelegate )
633  {
634  //TODO: Create RAII utility class for touch/release
635 
636  //Touch the control
637  Touch();
638 
639  //Send that token.
640  mAutomationDelegate->PostSetValueRequest(Identifier(), newNormalizedValue );
641 
642  //Release the control
643  Release();
644  }
645  else
646  {
647  mNeedNotify = true;
648 
649  // In the rare case that an automation delegate doesn't exist, lets still set the value. It's possible that someone is trying to
650  // set the new value before adding the parameter to a parametermanager.
651  UpdateNormalizedValue(newNormalizedValue);
652  }
653 }

655 template <typename T>
656 void AAX_CParameter<T>::UpdateNormalizedValue(double newNormalizedValue)
657 {
658  T newValue = mTaperDelegate->NormalizedToReal(newNormalizedValue);
659  if (mNeedNotify || (mValue.Get() != newValue))
660  {
661  //Set the new value
662  mValue.Set(newValue);
663 
664  //<DMT> Always notify that the value has changed through the automation delegate to guarantee that all control surfaces and other
665  // GUIs get their values updated.
666  if (mAutomationDelegate)
667  mAutomationDelegate->PostCurrentValue(Identifier(), newNormalizedValue);
668 
669  // clear flag
670  mNeedNotify = false;
671  }

674 template <typename T>
675 void AAX_CParameter<T>::InitializeNumberOfSteps()
676 {
677  if (mNumSteps == 0) // If no explicit number of steps has been set...
678  {
679  switch (mControlType)
680  {
681  case AAX_eParameterType_Discrete:
682  {
683  // Discrete parameters default to binary unless
684  // otherwise specified
685  this->SetNumberOfSteps (eParameterDefaultNumStepsDiscrete);
686  break;
687  }
688  case AAX_eParameterType_Continuous:
689  {
690  // Defaulting to 128 steps to match one full rotation of
691  // Command|8 and similar surfaces, which query the num
692  // steps to determine tick values for rotary encoders
693  this->SetNumberOfSteps (eParameterDefaultNumStepsContinuous);
694  break;
695  }
696  default:
697  {
698  AAX_ASSERT (0); // Invalid type
699  break;
700  }
701  }
702  }
